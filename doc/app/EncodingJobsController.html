def create
  @encoding_job = EncodingJob.new(user_params)
  if @encoding_job.save
    flash[:notice] = 'Created new encoding job'
    redirect_to encoding_jobs_path
  else
    flash[:warn] = "Unable to save encoding job. #{@encoding_job.errors.messages}"
    render :new
  end
end
