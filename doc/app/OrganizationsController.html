def update_multiple
  ids = params[:organization_ids] || []
  Organization.all.each do |o|
    allow_rw = (ids.include?(o.id.to_s) ? true : false)
    o.update_attribute(:can_write, allow_rw)
  end
  flash[:notice] = "Updated organizations."
  redirect_to organizations_path
end
